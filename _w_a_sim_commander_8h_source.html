96 /// Command data structure. The member contents depend on the command type as described in each command type of the `Enums::CommandId` enum documentation. \sa Enums::CommandId enum
97 struct WSMCMND_API Command
98 {
99 uint32_t token; ///< A unique ID for this command instance. Echoed back by server in command responses. Optional use for Client implementations.
100 uint32_t uData; ///< DWORD command parameter value, meaning depends on the command being issued.
101 double fData; ///< double-precision floating point command parameter value, meaning depends on the command being issued.
102 WSE::CommandId commandId; ///< What to do. \sa CommandId
103 char sData[STRSZ_CMD] = {0}; ///< String command parameter value, meaning depends on the command being issued.
104 // 544/544 B (packed/unpacked), 8/16 B aligned
105
106 /// Default constructor with all parameters optional.
107 explicit Command(WSE::CommandId id = WSE::CommandId::None, uint32_t uData = 0, const char *str = nullptr, double fData = 0.0, int32_t token = 0) :
108 token(token), uData(uData), fData(fData), commandId(id)
109 {
110 if (str) setStringData(str);
111 }
112
113 void setStringData(const char *str) { setCharArrayValue(sData, STRSZ_CMD, str); } ///< Set the `sData` member using a const char array.
114
115 /// `ostream` operator for logging purposes
116 friend inline std::ostream& operator<<(std::ostream& os, const Command &c)
117 {
118 const char *cmdName = (size_t)c.commandId < WSE::CommandIdNames.size() ? WSE::CommandIdNames.at((size_t)c.commandId) : "Invalid";
119 return os << "Command{" << cmdName << "; token: " << c.token << "; uData: " << c.uData
120 << "; fData: " << std::fixed << std::setprecision(9) << c.fData << "; sData: " << std::quoted(c.sData) << '}';
121 }
122 };

125 /// Structure for variable value subscription requests. \sa WASimCommander:CommandId::Subscribe command.
126 struct WSMCMND_API DataRequest
127 {
128 uint32_t requestId; ///< Unique ID for the request, subsequent use of this ID overwrites any previous request definition (but size may not grow).
129 uint32_t valueSize; ///< Byte size of stored value; can also be one of the predefined DATA_TYPE_* constants. \sa WASimCommander::DATA_TYPE_INT8, etc
130 float deltaEpsilon = 0.0f; ///< Minimum change in numeric value required to trigger an update. The default of `0.0` is to send updates only if the value changes, but even on the smallest changes.
131 /// Setting this to some positive value can be especially useful for high-precision floating-point numbers which may fluctuate within an insignifcant range,
132 /// but may be used with any numeric value (for integer value types, only the integer part of the epsilon value is considered).
133 /// Conversely, to send data updates _every time_ the value is read, and skip any comparison check altogether, set this to a negative value like `-1.0`.
134 ///< \note For the positive epsilon settings to work, the `valueSize` must be set to one of the predefined `DATA_TYPE_*` constants.
135 uint32_t interval = 0; ///< How many `UpdatePeriod` period's should elapse between checks. eg. 500ms or 10 ticks.
136 /// Zero means to check at every `period`, `1` means every other `period`, etc.
137 WSE::UpdatePeriod period; ///< How often to read/calculate this value.
138 WSE::RequestType requestType; ///< Named variable or calculated value.
139 WSE::CalcResultType calcResultType; ///< Expected calculator result type.
140 uint8_t simVarIndex; ///< Some SimVars require an index for access, default is 0.
141 char varTypePrefix = 'L'; ///< Variable type prefix for named variables. Types: 'L' (local), 'A' (SimVar) and 'T' (Token, not an actual GaugeAPI prefix) are checked using respecitive GaugeAPI methods.
142 char nameOrCode[STRSZ_REQ] = {0}; ///< Variable name or full calculator string.
143 char unitName[STRSZ_UNIT] = {0}; ///< Unit name for named variables (optional to override variable's default units). Only 'L' and 'A' variable types support unit specifiers.
144 // 1088/1088 B (packed/unpacked), 8/16 B aligned
145
146 /// Constructor with required request ID, all other parameters optional. See member documentation for explanation of the homonymous parameters.
147 explicit DataRequest(
148 uint32_t requestId,
149 uint32_t valueSize = sizeof(double),
150 WSE::RequestType requestType = WSE::RequestType::Calculated,
151 WSE::CalcResultType calcResultType = WSE::CalcResultType::Double,
152 WSE::UpdatePeriod period = WSE::UpdatePeriod::Tick,
153 const char * nameOrCode = nullptr,
154 const char * unitName = nullptr
155 )
156 : requestId(requestId), valueSize(valueSize), period(period), requestType(requestType), calcResultType(calcResultType)
157 {
158 if (nameOrCode)
159 setNameOrCode(nameOrCode);
160 if (unitName)
161 setUnitName(unitName);
162 }
163
164 /// Constructs a request for a named variable (`requestType = RequestType::Named`) with optional update period, interval, and epsilon values.
165 explicit DataRequest(uint32_t requestId, char variableType, const char *variableName, uint32_t valueSize,
166 WSE::UpdatePeriod period = WSE::UpdatePeriod::Tick, uint32_t interval = 0, float deltaEpsilon = 0.0f) :
167 requestId(requestId), valueSize(valueSize), deltaEpsilon(deltaEpsilon), interval(interval), period(period), requestType(WSE::RequestType::Named), varTypePrefix(variableType)
168 {
169 if (variableName)
170 setNameOrCode(variableName);
171 }
172 /// Constructs a request for a named Simulator Variable (`requestType = RequestType::Named` and `varTypePrefix = 'A'`) with optional update period, interval, and epsilon values.
173 explicit DataRequest(uint32_t requestId, const char *simVarName, const char *unitName, uint8_t simVarIndex, uint32_t valueSize,
174 WSE::UpdatePeriod period = WSE::UpdatePeriod::Tick, uint32_t interval = 0, float deltaEpsilon = 0.0f) :
175 requestId(requestId), valueSize(valueSize), deltaEpsilon(deltaEpsilon), interval(interval), period(period), requestType(WSE::RequestType::Named), simVarIndex(simVarIndex), varTypePrefix('A')
176 {
177 if (simVarName)
178 setNameOrCode(simVarName);
179 if (unitName)
180 setUnitName(unitName);
181 }
182 /// Constructs a calculator code request (`requestType = RequestType::Calculated`) with optional update period, interval, and epsilon values.
183 explicit DataRequest(uint32_t requestId, WSE::CalcResultType resultType, const char *calculatorCode, uint32_t valueSize,
184 WSE::UpdatePeriod period = WSE::UpdatePeriod::Tick, uint32_t interval = 0, float deltaEpsilon = 0.0f) :
185 requestId(requestId), valueSize(valueSize), deltaEpsilon(deltaEpsilon), interval(interval), period(period), requestType(WSE::RequestType::Calculated), calcResultType(resultType)
186 {
187 if (calculatorCode)
188 setNameOrCode(calculatorCode);
189 }
190
191 void setNameOrCode(const char *name) { setCharArrayValue(nameOrCode, STRSZ_REQ, name); } ///< Set the `nameOrCode` member using a const char array.
192 void setUnitName(const char *name) { setCharArrayValue(unitName, STRSZ_UNIT, name); } ///< Set the `unitName` member using a const char array.
193
194 /// `ostream` operator for logging purposes
195 friend inline std::ostream& operator<<(std::ostream& os, const DataRequest &r)
196 {
197 const char *perName = (size_t)r.period < WSE::UpdatePeriodNames.size() ? WSE::UpdatePeriodNames.at((size_t)r.period) : "Invalid";
198 os << "DataRequest{" << r.requestId << "; size: " << r.valueSize << "; period: " << perName << "; interval: " << r.interval << "; deltaE: " << r.deltaEpsilon;
199 if (r.requestType == WSE::RequestType::None)
200 return os << "; type: None; }";
201 if (r.requestType == WSE::RequestType::Named)
202 return os << "; type: Named; name: " << std::quoted(r.nameOrCode) << "; unit: " << std::quoted(r.unitName) << "; varType: '" << r.varTypePrefix << "'; varIndex: " << r.simVarIndex << '}';
203 const char *typeName = (size_t)r.calcResultType < WSE::CalcResultTypeNames.size() ? WSE::CalcResultTypeNames.at((size_t)r.calcResultType) : "Invalid";
204 return os << "; type: Calculated; code: " << std::quoted(r.nameOrCode) << " resultType: " << typeName << '}';
205 }
206 };

209 /// Log record structure. \sa WASimCommander:CommandId::Log command.
210 struct WSMCMND_API LogRecord
211 {
212 time_t timestamp; ///< ms since epoch.
213 WSE::LogLevel level; ///< Message severity.
214 char message[STRSZ_LOG] = {0}; ///< The log message text.
215 // 1040/1040 B (packed/unpacked), 8/16 B aligned
216
217 /// Default constructor with all parameters optional.
218 explicit LogRecord(const WSE::LogLevel level = WSE::LogLevel::None, const char *msg = nullptr, const std::chrono::system_clock::time_point &tp = std::chrono::system_clock::now()) :
219 timestamp{std::chrono::duration_cast<std::chrono::milliseconds>(tp.time_since_epoch()).count()}, level{level}
220 {
221 if (msg) setMessage(msg);
222 }
223
224 void setMessage(const char *msg) { setCharArrayValue(message, STRSZ_LOG, msg); } ///< Set the `message` member using a const char array.
225
226 /// `ostream` operator for logging purposes
227 friend inline std::ostream& operator<<(std::ostream& os, const LogRecord &l)
228 {
229 const time_t secs = l.timestamp / 1000;
230 const tm *tim = std::localtime(&secs);
231 const char *levelName = (size_t)l.level < WSE::LogLevelNames.size() ? WSE::LogLevelNames.at((size_t)l.level) : "Invalid";
232 if (tim)
233 return os << "LogRecord{" << levelName << std::put_time(tim, " [(%z) %y-%m-%d %H:%M:%S.") << (l.timestamp - time_t(secs * 1000)) << "] " << l.message << '}';
234 return os << "LogRecord{" << levelName << " [" << l.timestamp << "] " << l.message << '}';
235 }
236 };
