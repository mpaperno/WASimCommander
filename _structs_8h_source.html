43 // A fixed size char array... since "a member of a managed class cannot be a standard array"
44 // ... but apparently it's ok to have to make your own "standard array?" Strange and peculiar.
45 // This is used so we can memcpy between native and managed versions of the same packed structs.
46 // Originally found as "inline_array template written by Mark Hall and improved upon by Shaun Miller"
47 // on an archived blog by one of the CLR developers, Bran Bray:
48 // https://web.archive.org/web/20160126134209/http://blogs.msdn.com/b/branbray/archive/2005/07/20/441099.aspx
49 //
50 // Also the enable_if size check fails with a syntax error no matter what I try... but it's correct code.
51 // So don't try to make a zero size one, OK?
52 template<uint32_t SIZE /*, typename = std::enable_if_t<SIZE, uint32_t>*/ >
53 [Runtime::CompilerServices::UnsafeValueType]
54 [StructLayout(LayoutKind::Explicit, Size=(sizeof(Byte)*SIZE))]
55 public value struct char_array {
56 private:
57 [FieldOffset(0)]
58 Byte _item;
59
60 public:
61 char_array<SIZE>(String ^str) { fromManagedStr(str); }
62 char_array<SIZE>(const std::string &str) { fromStdStr(str); }
63 char_array<SIZE>(const char *str) : char_array<SIZE>(str, std::strlen(str)) {}
64 char_array<SIZE>(const char *str, const size_t len) {
65 clear();
66 const pin_ptr<Byte> ptr = &_item;
67 memcpy(ptr, str, min(SIZE, len));
68 }
69
70 void fromStdStr(const std::string &str) {
71 clear();
72 if (str.empty())
73 return;
74 const pin_ptr<Byte> ptr = &_item;
75 const uint32_t len = std::min<uint32_t>(SIZE-1, (uint32_t)str.size());
76 memcpy(ptr, str.data(), len);
77 //if (len < SIZE)
78 //memset(ptr + len, 0, (size_t)SIZE - len);
79 }

81 void fromManagedStr(String ^str) {
82 clear();
83 if (String::IsNullOrEmpty(str))
84 return;
85 const pin_ptr<Byte> ptr = &_item;
86 const array<Byte> ^strData = Text::Encoding::UTF8->GetBytes(str);
87 const pin_ptr<Byte> src = &strData[0];
88 const uint32_t len = std::min<uint32_t>(SIZE-1, strData->Length);
89 memcpy(ptr, src, len);
90 //if (len < SIZE)
91 //memset(ptr + len, 0, (size_t)SIZE - len);
92 }
93
94 void clear() {
95 const pin_ptr<Byte> ptr = &_item;
96 memset(ptr, 0, SIZE);
97 }

294 /// <summary> `DataRequestRecord` inherits and extends `WASimCommander::CLI::Structs::DataRequest` with data pertinent for use by a data consumer/Client.
295 /// In particular, any value data sent from the server is stored here as a byte array in the `data` member (a byte array). </summary>
296 ///
297 /// `DataRequest` subscription results are delivered by `WASimClient` (via `OnDataReceived` event) using this `DataRequestRecord` structure.
298 /// WASimClient also holds a list of all data requests which have been added using `WASimClient::saveDataRequest()` method. These
299 /// requests are available for reference using `WASimClient::dataRequest()`, `WASimClient::dataRequests()`, and `WASimClient::dataRequestIdsList()` methods.
300 /// For full documentation of each field see `WASimCommander::Client::DataRequestRecord`.
301 ///
302 /// For assistance with converting the raw data to actual values, there are a couple `bool tryConvert(T)` templates which try to populate a value reference
303 /// (`out` parameter) of the desired type and returns true or false depending on if the conversion was valid. Validity is based only on the data array size vs.
304 /// the requested type size, so it will happily populate an Int32 with data representing a float value, since the type sizes match. **The user must know what data
305 /// type they are expecting.**
306 ///
307 /// One version of `tryConvert()` handles basic numeric types, and another is a special overload for returning Strings.
308 ///
309 /// There are also implicit conversion operators for all numeric types and String. If conversion fails, these return a default-constructed instance,
310 /// typically zero or an empty string.
311 ///
312 /// Also note that the size of the data array may or _may not_ be the same as the inherited `DataRequest::valueSize` member, since that may contain
313 /// special values for preset types. Always check the actual data array size (`data.Length`) if you need to know the storage requirements.
314 [StructLayout(LayoutKind::Sequential, CharSet = CharSet::Ansi, Pack = 1)]
315 public ref struct DataRequestRecord : public DataRequest
316 {
317 public:
318 time_t lastUpdate = 0; ///< Timestamp of last data update in ms since epoch.
319 array<Byte> ^data {}; ///< Value data array.
320
321
322 /// <summary> Tries to populate a value reference of the desired type and returns true or false
323 /// depending on if the conversion was valid (meaning the size of requested type matches the data size). </summary>
324 /// If the conversion fails, result is default-initialized.
325 /// The requested type must be a `value` type (not reference) and be default-constructible, (eg. numerics, chars), or fixed-size arrays of such types.
326 generic<typename T> where T : value class, gcnew()
327 inline bool tryConvert([Out] T %result)
328 {
329 if (data->Length == (int)sizeof(T)) {
330 pin_ptr<T> pr = &result;
331 pin_ptr<Byte> pd = &data[0];
332 memcpy_s(pr, sizeof(T), pd, data->Length);
333 return true;
334 }
335 result = T();
336 return false;
337 }
338
339 /// <summary> Overload for String type results. This will simply load whatever bytes are in the `data` array into a String, and only
340 /// returns `false` (and an empty String) if the data array is empty or starts with a null character byte (`0x00`). </summary>
341 inline bool tryConvert([Out] String ^ %result)
342 {
343 if (!data->Length || !data[0]) {
344 result = String::Empty;
345 return false;
346 }
347 pin_ptr<Byte> pd = &data[0];
348 result = gcnew String((const char *)pd);
349 return true;
350 }
351
352 // Implicit conversion operators for various types
353 inline static operator double(DataRequestRecord ^dr) { return dr->toType<double>(); }
354 inline static operator float(DataRequestRecord ^dr) { return dr->toType<float>(); }
355 inline static operator int64_t(DataRequestRecord ^dr) { return dr->toType<int64_t>(); }
356 inline static operator uint64_t(DataRequestRecord ^dr) { return dr->toType<uint64_t>(); }
357 inline static operator int32_t(DataRequestRecord ^dr) { return dr->toType<int32_t>(); }
358 inline static operator uint32_t(DataRequestRecord ^dr) { return dr->toType<uint32_t>(); }
359 inline static operator int16_t(DataRequestRecord ^dr) { return dr->toType<int16_t>(); }
360 inline static operator uint16_t(DataRequestRecord ^dr) { return dr->toType<uint16_t>(); }
361 inline static operator int8_t(DataRequestRecord ^dr) { return dr->toType<int8_t>(); }
362 inline static operator uint8_t(DataRequestRecord ^dr) { return dr->toType<uint8_t>(); }
363 inline static operator String ^(DataRequestRecord ^dr) { return dr->toType<String ^>(); }

376 return String::Format(
377 "{0}; DataRequestRecord {{Last Update: {1}; Data: {2}}}",

379 DateTimeOffset::FromUnixTimeMilliseconds(lastUpdate).LocalDateTime.ToString("HH:mm:ss.fff"),

386 DataRequest((WASimCommander::DataRequest)dr), lastUpdate{dr.lastUpdate}, data(gcnew array<Byte>((int)dr.data.size()))
387 {

389 }

424 /// <summary> Structure for delivering list results, eg. of local variables sent from Server. </summary>
425 /// For full documentation of each field see `WASimCommander::Client::ListResult`.
426 public ref struct ListResult sealed
427 {
428 public:
429 using ListCollectionType = Dictionary<int, String^>;
430
431 LookupItemType listType;
432 HRESULT result;
433 ListCollectionType ^list;
434
435 String ^ToString() override {
436 Text::StringBuilder sb(String::Format("ListResult {{{0}; result: {1}; Values:\n", listType, result));
437 ListCollectionType::Enumerator en = list->GetEnumerator();
438 while (en.MoveNext())
439 sb.AppendFormat("{0,4} = {1}\n", en.Current.Key, en.Current.Value);
440 sb.Append("}");
441 return sb.ToString();
442 }

446 listType{(LookupItemType)r.listType}, result(r.result), list{gcnew ListCollectionType((int)r.list.size()) }
447 {
448 for (const auto &pr : r.list)
449 list->Add(pr.first, gcnew String(pr.second.c_str()));
450 }
