127 struct TrackedRequest : public DataRequest
128 {
129 uint32_t dataId; // our area and def ID
130 time_t lastUpdate = 0;
131 uint32_t dataSize;
132 mutable shared_mutex m_dataMutex;
133 vector<uint8_t> data;
134
135 explicit TrackedRequest(const DataRequest &req, uint32_t dataId) :
136 DataRequest(req),
137 dataId{dataId},
138 dataSize{Utilities::getActualValueSize(valueSize)},
139 data(dataSize, 0xFF)
140 { }
141
142 TrackedRequest & operator=(const DataRequest &req) {
143 if (req.valueSize != valueSize) {
144 unique_lock lock(m_dataMutex);
145 dataSize = Utilities::getActualValueSize(req.valueSize);
146 data = vector<uint8_t>(dataSize, 0xFF);
147 }
148 DataRequest::operator=(req);
149 return *this;
150 }
151
152 DataRequestRecord toRequestRecord() const {
153 DataRequestRecord drr = DataRequestRecord(static_cast<const DataRequest &>(*this));
154 drr.data.assign(data.cbegin(), data.cend());
155 drr.lastUpdate = lastUpdate;
156 return drr;
157 }

274 // Rvalue arguments version
275 template<typename... Args>
276 void invokeCallbackImpl(function<void(Args...)> f, Args&&... args) const
277 {
278 if (f) {
279 try {
280#if defined(WSMCMND_CLIENT_USE_CONCURRENT_CALLBACKS) && !WSMCMND_CLIENT_USE_CONCURRENT_CALLBACKS
281 lock_guard lock(mtxCallbacks);
282#endif
283 bind(f, forward<Args>(args)...)();
284 }
285 catch (exception *e) { LOG_ERR << "Exception in callback handler: " << e->what(); }
286 }
287 }
288
289 // const Lvalue arguments version
290 template<typename... Args>
291 void invokeCallbackImpl(function<void(const Args...)> f, const Args... args) const
292 {
293 if (f) {
294 try {
295#if defined(WSMCMND_CLIENT_USE_CONCURRENT_CALLBACKS) && !WSMCMND_CLIENT_USE_CONCURRENT_CALLBACKS
296 lock_guard lock(mtxCallbacks);
297#endif
298 bind(f, args...)();
299 }
300 catch (exception *e) { LOG_ERR << "Exception in callback handler: " << e->what(); }
301 }
302 }

305 template<typename F, typename... Args>
306 void invokeCallback(F&& f, Args&&... args) const {
307 invokeCallbackImpl<remove_cv_t<remove_reference_t<Args>>...>(forward<F>(f), forward<Args>(args)...);
308 }

384 void setStatus(ClientStatus s, const string &msg = "")
385 {
386 static const char * evTypeNames[] = {
387 "None",
388 "Simulator Connecting", "Simulator Connected", "Simulator Disconnecting", "Simulator Disconnected",
389 "Server Connecting", "Server Connected", "Server Disconnected"
390 };
391
392 ClientStatus newStat = status;
393 ClientEvent ev;
394 switch (s) {
395 case ClientStatus::Idle:
396 ev.eventType = ClientEventType::SimDisconnected;
397 newStat = s;
398 break;
399 case ClientStatus::Initializing:
400 ev.eventType = (status & ClientStatus::SimConnected) == ClientStatus::SimConnected ? ClientEventType::SimDisconnected : ClientEventType::SimConnecting;
401 newStat = s;
402 break;
403 case ClientStatus::SimConnected:
404 ev.eventType = (status & ClientStatus::Connected) == ClientStatus::Connected ? ClientEventType::ServerDisconnected : ClientEventType::SimConnected;
405 newStat = (newStat & ~(ClientStatus::Connected | ClientStatus::Connecting | ClientStatus::Initializing)) | s;
406 break;
407 case ClientStatus::Connecting:
408 ev.eventType = ClientEventType::ServerConnecting;
409 newStat = (newStat & ~ClientStatus::Connected) | s;
410 break;
411 case ClientStatus::Connected:
412 ev.eventType = ClientEventType::ServerConnected;
413 newStat = (newStat & ~ClientStatus::Connecting) | s;
414 break;
415 case ClientStatus::ShuttingDown:
416 ev.eventType = ClientEventType::SimDisconnecting;
417 newStat |= s;
418 break;
419 }
420 if (newStat == status)
421 return;
422 status = newStat;
423 if (eventCb) {
424 ev.status = status;
425 if (msg == "")
426 ev.message = Utilities::getEnumName(ev.eventType, evTypeNames);
427 else
428 ev.message = msg;
429 invokeCallback(eventCb, move(ev));
430 }
431 }

433 // simple thread burner loop to wait on a condition... don't abuse.
434 bool waitCondition(const std::function<bool(void)> predicate, uint32_t timeout, uint32_t sleepMs = 1)
435 {
436 const auto endTime = Clock::now() + chrono::milliseconds(timeout);
437 const auto sf = chrono::milliseconds(sleepMs);
438 while (!predicate() && endTime > Clock::now())
439 this_thread::sleep_for(sf);
440 return predicate();
441 }

518 HRESULT connectSimulator(int networkConfigId, uint32_t timeout)
519 {
520 if ((status & ClientStatus::Initializing) == ClientStatus::Initializing)
521 return E_FAIL; // unlikely, but prevent recursion
522
523 if (checkInit()) {
524 LOG_INF << WSMCMND_CLIENT_NAME " already initialized.";
525 return E_FAIL;
526 }
527 settings.networkConfigId = networkConfigId;
528 LOG_INF << "Initializing " << WSMCMND_CLIENT_NAME << " v" WSMCMND_VERSION_INFO " for client " << clientName << " with net config ID " << settings.networkConfigId;
529 setStatus(ClientStatus::Initializing);
530
531 if (!timeout)
532 timeout = settings.networkTimeout;
533
534 if (!hDispatchStopEvent) {
535 hDispatchStopEvent = CreateEvent(nullptr, true, false, nullptr);
536 if (!hDispatchStopEvent) {
537 LOG_ERR << "Failed to CreateEvent() for dispatch loop: " << GetLastError();
538 return E_FAIL;
539 }
540 }
541
542 hSimEvent = CreateEvent(nullptr, false, false, nullptr);
543 simConnected = false;
544
545 // try to open connection; may return S_OK, E_FAIL, or E_INVALIDARG. May also time out and never actually send the Open event message.
546 HRESULT hr = SimConnect_Open(&hSim, clientName.c_str(), nullptr, 0, hSimEvent, settings.networkConfigId);
547
548 if FAILED(hr) {
549 LOG_ERR << "Network connection to Simulator failed with result: " << LOG_HR(hr);
550 hSim = nullptr;
551 setStatus(ClientStatus::Idle);
552 return hr;
553 }
554 // start message dispatch loop. required to do this now in order to receive the SimConnect Open event and finish the connection process.
555 ResetEvent(hDispatchStopEvent);
556 dispatchThread = thread(&Private::dispatchLoop, this);
557 dispatchThread.detach();
558 // wait for dispatch loop to start
559 if (!waitCondition([&]() { return !!runDispatchLoop; }, 10)) {
560 LOG_CRT << "Could not start dispatch loop thread.";
561 disconnectSimulator();
562 return E_FAIL;
563 }
564 // wait until sim connects or times out or disconnectSimulator() is called.
565 if (!waitCondition([&]() { return !runDispatchLoop || simConnected; }, timeout)) {
566 LOG_ERR << "Network connection to Simulator timed out after " << timeout << "ms.";
567 disconnectSimulator();
568 return E_TIMEOUT;
569 }
570
571 // register server's Connect command event, exit on failure
572 if FAILED(hr = SimConnectHelper::newClientEvent(hSim, CLI_EVENT_CONNECT, EVENT_NAME_CONNECT)) {
573 disconnectSimulator();
574 return hr;
575 }
576 // register server's ping command event
577 SimConnectHelper::newClientEvent(hSim, CLI_EVENT_PING, EVENT_NAME_PING);
578 // register and subscribe to server ping response (the server will append the name of our client to a new named event and trigger it).
579 SimConnectHelper::newClientEvent(hSim, CLI_EVENT_PING_RESP, EVENT_NAME_PING_PFX + clientName, NOTIFY_GROUP_PING, SIMCONNECT_GROUP_PRIORITY_HIGHEST);
580
581 // register area for writing commands (this is read-only for the server)
582 if FAILED(hr = registerDataArea(CDA_NAME_CMD_PFX, CLI_DATA_COMMAND, CLI_DATA_COMMAND, sizeof(Command), true)) {
583 disconnectSimulator();
584 return hr;
585 }
586 // register command response area for reading (server can write to this channel)
587 if FAILED(hr = registerDataArea(CDA_NAME_RESP_PFX, CLI_DATA_RESPONSE, CLI_DATA_RESPONSE, sizeof(Command), false)) {
588 disconnectSimulator();
589 return hr;
590 }
591 // register CDA for writing data requests (this is read-only for the server)
592 registerDataArea(CDA_NAME_DATA_PFX, CLI_DATA_REQUEST, CLI_DATA_REQUEST, sizeof(DataRequest), true);
593 // register CDA for writing key events (this is read-only for the server)
594 registerDataArea(CDA_NAME_KEYEV_PFX, CLI_DATA_KEYEVENT, CLI_DATA_KEYEVENT, sizeof(KeyEvent), true);
595
596 // start listening on the response channel
597 if FAILED(hr = INVOKE_SIMCONNECT(
598 RequestClientData, hSim, (SIMCONNECT_CLIENT_DATA_ID)CLI_DATA_RESPONSE, (SIMCONNECT_DATA_REQUEST_ID)DATA_REQ_RESPONSE,
599 (SIMCONNECT_CLIENT_DATA_DEFINITION_ID)CLI_DATA_RESPONSE, SIMCONNECT_CLIENT_DATA_PERIOD_ON_SET, (SIMCONNECT_CLIENT_DATA_REQUEST_FLAG)0, 0UL, 0UL, 0UL
600 )) {
601 disconnectSimulator();
602 return hr;
603 }
604
605 // possibly re-register SimConnect CDAs for any existing data requests
606 registerAllDataRequestAreas();
607 // re-register all Simulator Custom Events
608 mapAllCustomKeyEvents();
609
610 setStatus(ClientStatus::SimConnected);
611 return S_OK;
612 }

614 // onQuitEvent == true when SimConnect is shutting down and sends Quit, in which case we don't try to use SimConnect again.
615 void disconnectSimulator(bool onQuitEvent = false)
616 {
617 if (!checkInit()) {
618 LOG_INF << "No network connection. Shutdown complete.";
619 setStatus(ClientStatus::Idle);
620 return;
621 }
622
623 LOG_DBG << "Shutting down...";
624 disconnectServer(!onQuitEvent);
625 setStatus(ClientStatus::ShuttingDown);
626 SetEvent(hDispatchStopEvent); // signal thread to exit
627 //d->dispatchThread.join();
628
629 // check for dispatch loop thread shutdown (this is really just for logging/debug purposes)
630 if (!waitCondition([&]() { return !runDispatchLoop; }, DISPATCH_LOOP_WAIT_TIME + 100))
631 LOG_CRT << "Dispatch loop thread still running!";
632
633 // reset flags/counters
634 simConnected = false;
635 logCDAcreated = false;
636 totalDataAlloc = 0;
637
638 // dispose objects
639 if (hSim && !onQuitEvent)
640 SimConnect_Close(hSim);
641 hSim = nullptr;
642 if (hSimEvent)
643 CloseHandle(hSimEvent);

650 HRESULT connectServer(uint32_t timeout)
651 {
652 if ((status & ClientStatus::Connecting) == ClientStatus::Connecting)
653 return E_FAIL; // unlikely, but prevent recursion
654
655 if (!timeout)
656 timeout = settings.networkTimeout;
657
658 HRESULT hr;
659 if (!checkInit() && FAILED(hr = connectSimulator(timeout)))
660 return hr;
661
662 LOG_INF << "Connecting to " WSMCMND_PROJECT_NAME " server...";
663 setStatus(ClientStatus::Connecting);
664
665 // Send the initial "connect" event which we registered in connectSimulator(), using this client's ID as the data parameter.
666 if FAILED(hr = INVOKE_SIMCONNECT(TransmitClientEvent, hSim, SIMCONNECT_OBJECT_ID_USER, (SIMCONNECT_CLIENT_EVENT_ID)CLI_EVENT_CONNECT, (DWORD)clientId, SIMCONNECT_GROUP_PRIORITY_HIGHEST, SIMCONNECT_EVENT_FLAG_GROUPID_IS_PRIORITY)) {
667 setStatus(ClientStatus::SimConnected);
668 return hr;
669 }
670 // The connection response, if any, is written by the server into the "command response" client data area, CLI_DATA_ID_RESPONSE, processed in SIMCONNECT_RECV_ID_CLIENT_DATA
671
672 // wait until server connects or times out or disconnectSimulator() is called.
673 if (!waitCondition([&]() { return !runDispatchLoop || serverConnected; }, timeout)) {
674 LOG_ERR << "Server connection timed out or refused after " << timeout << "ms.";
675 setStatus(ClientStatus::SimConnected);
676 return E_TIMEOUT;
677 }
678
679 LOG_INF << "Connected to " WSMCMND_PROJECT_NAME " server v" << STREAM_HEX8(serverVersion);
680 if ((serverVersion & 0xFF000000) != (WSMCMND_VERSION & 0xFF000000))
681 LOG_WRN << "Server major version does not match WASimClient version " << STREAM_HEX8(WSMCMND_VERSION);
682 setStatus(ClientStatus::Connected);
683
684 // clear any pending list request (unlikely)
685 listResult.reset();
686 // make sure server knows our desired log level and set up data area/request if needed
687 updateServerLogLevel();
688 // set update status of data requests before adding any, in case we don't actually want results yet
689 sendServerCommand(Command(CommandId::Subscribe, (requestsPaused ? 0 : 1)));
690 // (re-)register (or delete) any saved DataRequests
691 registerAllDataRequests();
692 // same with calculator events
693 registerAllEvents();
694
695 return S_OK;
696 }

798 // Blocks and waits for a response to a specific command token. `timeout` can be -1 to use `extraPredicate` only (which is then required).
799 HRESULT waitCommandResponse(uint32_t token, Command *response, uint32_t timeout = 0, std::function<bool(void)> extraPredicate = nullptr)
800 {
801 TrackedResponse *tr = findTrackedResponse(token);
802 shared_ptr<condition_variable_any> cv {};
803 if (!tr || !(cv = tr->cv.lock()))
804 return E_FAIL;
805
806 if (!timeout)
807 timeout = settings.networkTimeout;
808
809 bool stopped = false;
810 auto stop_waiting = [=, &stopped]() {
811 return stopped =
812 !serverConnected || (tr->response.commandId != CommandId::None && tr->response.token == token && (!extraPredicate || extraPredicate()));
813 };
814
815 HRESULT hr = E_TIMEOUT;
816 if (stop_waiting()) {
817 hr = S_OK;
818 }
819 else {
820 unique_lock lock(tr->mutex);
821 if (timeout > 0) {
822 if (cv->wait_for(lock, chrono::milliseconds(timeout), stop_waiting))
823 hr = S_OK;
824 }
825 else if (!!extraPredicate) {
826 cv->wait(lock, stop_waiting);
827 hr = stopped ? ERROR_ABANDONED_WAIT_0 : S_OK;
828 }
829 else {
830 hr = E_INVALIDARG;
831 LOG_DBG << "waitCommandResponse() requires a predicate condition when timeout parameter is < 0.";
832 }
833 }
834 if (SUCCEEDED(hr) && !!response) {
835 unique_lock lock(tr->mutex);
836 *response = move(tr->response);
837 }
838
839 unique_lock vlock(mtxResponses);
840 reponses.erase(token);
841 return hr;
842 }

1104 HRESULT addOrUpdateRequest(const DataRequest &req, bool async)
1105 {
1106 if (req.requestType == RequestType::None)
1107 return removeRequest(req.requestId);
1108
1109 // preliminary validation
1110 if (req.nameOrCode[0] == '\0') {
1111 LOG_ERR << "Error in DataRequest ID: " << req.requestId << "; Parameter 'nameOrCode' cannot be empty.";
1112 return E_INVALIDARG;
1113 }
1114
1115 const uint32_t actualValSize = Utilities::getActualValueSize(req.valueSize);
1116 if (actualValSize > SIMCONNECT_CLIENTDATA_MAX_SIZE) {
1117 LOG_ERR << "Error in DataRequest ID: " << req.requestId << "; Value size " << actualValSize << " exceeds SimConnect maximum size " << SIMCONNECT_CLIENTDATA_MAX_SIZE;
1118 return E_INVALIDARG;
1119 }
1120 if (totalDataAlloc + actualValSize > SIMCONNECT_DATA_ALLOC_LIMIT) {
1121 LOG_ERR << "Error in DataRequest ID: " << req.requestId << "; Adding request with value size " << actualValSize << " would exceed SimConnect total maximum size of " << SIMCONNECT_DATA_ALLOC_LIMIT;
1122 return E_INVALIDARG;
1123 }
1124
1125 TrackedRequest *tr = findRequest(req.requestId);
1126 const bool isNewRequest = (tr == nullptr);
1127 bool dataAllocationChanged = false;
1128
1129 if (isNewRequest) {
1130 unique_lock lock{mtxRequests};
1131 tr = &requests.try_emplace(req.requestId, req, nextDefId++).first->second;
1132 }
1133 else {
1134 if (actualValSize > tr->dataSize) {
1135 LOG_ERR << "Value size cannot be increased after request is created.";
1136 return E_INVALIDARG;
1137 }
1138 // flag if the definition size or the delta E has changed
1139 dataAllocationChanged = (actualValSize != tr->dataSize || !fuzzyCompare(req.deltaEpsilon, tr->deltaEpsilon));
1140 // update the tracked request from new request data
1141 unique_lock lock{mtxRequests};
1142 *tr = req;
1143 }
1144
1145 HRESULT hr = S_OK;
1146 if (isConnected()) {
1147 // Register the CDA and subscribe to data value changes
1148 if (isNewRequest || dataAllocationChanged)
1149 hr = registerDataRequestArea(tr, isNewRequest, dataAllocationChanged);
1150 if SUCCEEDED(hr) {
1151 // send the request and wait for Ack; Request may timeout or return a Nak.
1152 hr = sendDataRequest(req, async);
1153 }
1154 if (FAILED(hr) && isNewRequest) {
1155 // delete a new request if anything failed
1156 removeRequest(req.requestId);
1157 }
1158 LOG_TRC << (FAILED(hr) ? "FAILED" : isNewRequest ? "Added" : "Updated") << " request: " << req;
1159 }
1160 else {
1161 LOG_TRC << "Queued " << (isNewRequest ? "New" : "Updated") << " request: " << req;
1162 }
1163
1164 return hr;
1165 }

1354 void dispatchLoop()
1355 {
1356 LOG_DBG << "Dispatch loop started.";
1357 const HANDLE waitEvents[] = { hDispatchStopEvent, hSimEvent };
1358 DWORD hr;
1359 runDispatchLoop = true;
1360 while (runDispatchLoop) {
1361 hr = WaitForMultipleObjects(2, waitEvents, false, DISPATCH_LOOP_WAIT_TIME);
1362 switch (hr) {
1363 case WAIT_TIMEOUT:
1364 continue;
1365 case WAIT_OBJECT_0 + 1: // hSimEvent
1366 SimConnect_CallDispatch(hSim, Private::dispatchMessage, this);
1367 continue;
1368 case WAIT_OBJECT_0: // hDispatchStopEvent
1369 break;
1370 default:
1371 LOG_ERR << "Dispatch loop WaitForMultipleObjects returned: " << LOG_HR(hr) << " Error: " << LOG_HR(GetLastError());
1372 break;
1373 }
1374 runDispatchLoop = false;
1375 }
1376 LOG_DBG << "Dispatch loop stopped.";
1377 }
1378
1379 void onSimConnectMessage(SIMCONNECT_RECV *pData, DWORD cbData)
1380 {
1381 LOG_TRC << LOG_SC_RECV(pData);
1382 switch (pData->dwID) {
1383
1384 case SIMCONNECT_RECV_ID_CLIENT_DATA: {
1385 SIMCONNECT_RECV_CLIENT_DATA* data = (SIMCONNECT_RECV_CLIENT_DATA*)pData;
1386 LOG_TRC << LOG_SC_RCV_CLIENT_DATA(data);
1387 // dwSize always under-reports by 4 bytes when sizeof(SIMCONNECT_RECV_CLIENT_DATA) is subtracted, and the minimum reported size is 4 bytes even for 0-3 bytes of actual data.
1388 const size_t dataSize = (size_t)pData->dwSize + 4 - sizeof(SIMCONNECT_RECV_CLIENT_DATA);
1389 switch (data->dwRequestID)
1390 {
1391 case DATA_REQ_RESPONSE: {
1392 // be paranoid
1393 if (dataSize != sizeof(Command)) {
1394 LOG_CRT << "Invalid Command struct data size! Expected " << sizeof(Command) << " but got " << dataSize;
1395 return;
1396 }
1397 // dwData is our command struct
1398 const Command *const cmd = reinterpret_cast<const Command *const>(&data->dwData);
1399 LOG_DBG << "Got Command: " << *cmd;
1400 bool checkTracking = true;
1401 switch (cmd->commandId)
1402 {
1403 case CommandId::Ack:
1404 case CommandId::Nak: {
1405 // which command is this ack/nak for?
1406 switch ((CommandId)cmd->uData) {
1407 // Connected response
1408 case CommandId::Connect:
1409 serverConnected = cmd->commandId == CommandId::Ack && cmd->token == clientId;
1410 serverVersion = (uint32_t)cmd->fData;
1411 checkTracking = false;
1412 break;
1413 default:
1414 break;
1415 }
1416 // invoke result callback if anyone is listening
1417 invokeCallback(cmdResultCb, *cmd);
1418 break;
1419 }
1420
1421 // respond to pings to stay alive
1422 case CommandId::Ping:
1423 sendServerCommand(Command(CommandId::Ack, (uint32_t)CommandId::Ping));
1424 checkTracking = false;
1425 break;
1426
1427 // incoming list results, one at a time until Ack of List command.
1428 case CommandId::List: {
1429 unique_lock lock(listResult.mutex);
1430 if (cmd->token == listResult.token) {
1431 listResult.nextTimeout = Clock::now() + chrono::milliseconds(settings.networkTimeout);
1432 listResult.result.emplace_back((int)cmd->uData, cmd->sData);
1433 }
1434 else {
1435 LOG_WRN << "Received unexpected list result for wrong command token. Expected " << listResult.token << " got " << cmd->token;
1436 }
1437 checkTracking = false;
1438 break;
1439 }
1440
1441 // Server is disconnecting (shutting down/etc).
1442 case CommandId::Disconnect:
1443 disconnectServer(false);
1444 checkTracking = false;
1445 break;
1446
1447 default:
1448 break;
1449 } // commandId
1450
1451 // Check if this command response is tracked and possibly awaited.
1452 if (checkTracking) {
1453 if (TrackedResponse *tr = findTrackedResponse(cmd->token)) {
1454 if (auto cv = tr->cv.lock()) {
1455 LOG_TRC << "Got awaited command response, notifying CV.";
1456 //unique_lock lock(mtxResponses);
1457 unique_lock lock(tr->mutex);
1458 tr->response = *cmd;
1459 lock.unlock();
1460 cv->notify_all();
1461 }
1462 else {
1463 unique_lock lock(mtxResponses);
1464 reponses.erase(cmd->token);
1465 LOG_TRC << "Got tracked command response but no CV, removing record.";
1466 }
1467 }
1468 }
1469 // invoke response callback if anyone is listening
1470 invokeCallback(respCb, *cmd);
1471 break;
1472 } // DATA_REQ_RESPONSE
1473
1474 case DATA_REQ_LOG: {
1475 // be paranoid
1476 if (dataSize != sizeof(LogRecord)) {
1477 LOG_CRT << "Invalid LogRecord struct data size! Expected " << sizeof(LogRecord) << " but got " << dataSize;
1478 return;
1479 }
1480 // dwData is our log record struct
1481 const LogRecord *const log = reinterpret_cast<const LogRecord *const>(&data->dwData);
1482 LOG_TRC << "Got Log Record: " << *log;
1483 invokeCallback(logCb, *log, LogSource::Server);
1484 break;
1485 } // DATA_REQ_LOG
1486
1487 // possible request data
1488 default:
1489 if (data->dwRequestID >= SIMCONNECTID_LAST) {
1490 TrackedRequest *tr = findRequest(data->dwRequestID - SIMCONNECTID_LAST);
1491 if (!tr){
1492 LOG_WRN << "DataRequest ID " << data->dwRequestID - SIMCONNECTID_LAST << " not found in tracked requests.";
1493 return;
1494 }
1495 // be paranoid; note that the reported pData->dwSize is never less than 4 bytes.
1496 if (dataSize < tr->dataSize) {
1497 LOG_CRT << "Invalid data result size! Expected " << tr->dataSize << " but got " << dataSize;
1498 return;
1499 }
1500 unique_lock datalock(tr->m_dataMutex);
1501 memcpy(tr->data.data(), (void*)&data->dwData, tr->dataSize);
1502 tr->lastUpdate = chrono::duration_cast<chrono::milliseconds>(chrono::system_clock::now().time_since_epoch()).count();
1503 datalock.unlock();
1504 shared_lock rdlock(mtxRequests);
1505 LOG_TRC << "Got data result for request: " << *tr;
1506 invokeCallback(dataCb, tr->toRequestRecord());
1507 break;
1508 }
1509 LOG_WRN << "Got unknown RequestID in SIMCONNECT_RECV_CLIENT_DATA struct: " << data->dwRequestID;
1510 return;
1511 } // data->dwRequestID
1512
1513 //serverLastSeen = Clock::now();
1514 break;
1515 } // SIMCONNECT_RECV_ID_CLIENT_DATA
1516
1517 case SIMCONNECT_RECV_ID_EVENT: {
1518 SIMCONNECT_RECV_EVENT *data = static_cast<SIMCONNECT_RECV_EVENT *>(pData);
1519 LOG_TRC << LOG_SC_RECV_EVENT(data);
1520
1521 switch (data->uEventID) {
1522 case CLI_EVENT_PING_RESP:
1523 serverVersion = data->dwData;
1524 serverLastSeen = Clock::now();
1525 LOG_DBG << "Got ping response at " << Utilities::timePointToString(serverLastSeen.load()) << " with version " << STREAM_HEX8(serverVersion);
1526 break;
1527
1528 default:
1529 return;
1530 }
1531 break;
1532 } // SIMCONNECT_RECV_ID_EVENT
1533
1534 case SIMCONNECT_RECV_ID_OPEN:
1535 simConnected = true;
1536 SimConnectHelper::logSimVersion(pData);
1537 break;
1538
1539 case SIMCONNECT_RECV_ID_QUIT: {
1540 LOG_INF << "Received quit command from SimConnect, disconnecting...";
1541 // we're being called from within the message dispatch loop thread; to avoid deadlock we must disconnect asynchronously.
1542 thread([=]() { disconnectSimulator(true); }).detach();
1543 break;
1544 }
1545
1546 case SIMCONNECT_RECV_ID_EXCEPTION:
1547 SimConnectHelper::logSimConnectException(pData);
1548 break;
1549
1550 default:
1551 break;
1552 }
1553 } // onSimConnectMessage()

1607{
1608 if (!d->checkInit() && FAILED(d->connectSimulator(timeout)))
1609 return 0;
1610
1611 // save the last time we saw the server
1612 const Clock::time_point refTp = d->serverLastSeen;
1613 // Send the previously-registered PING event with this client's ID as the data parameter.
1614 if FAILED(INVOKE_SIMCONNECT(
1615 TransmitClientEvent, d->hSim, SIMCONNECT_OBJECT_ID_USER, (SIMCONNECT_CLIENT_EVENT_ID)Private::CLI_EVENT_PING, (DWORD)d->clientId, SIMCONNECT_GROUP_PRIORITY_HIGHEST, SIMCONNECT_EVENT_FLAG_GROUPID_IS_PRIORITY
1616 ))
1617 return 0;
1618
1619 if (!timeout)
1620 timeout = defaultTimeout();
1621 // The ping response (if any) should come back via the CLI_EVENT_PING_RESP event which we've subscribed to (SIMCONNECT_RECV_ID_EVENT).
1622 if (d->waitCondition([&]() { return !d->runDispatchLoop || d->serverLastSeen.load() > refTp; }, timeout)) {
1623 LOG_INF << "Server responded to ping, reported version: " << STREAM_HEX8(d->serverVersion);
1624 return d->serverVersion;
1625 }
1626 LOG_WRN << "Server did not respond to ping after " << timeout << "ms";
1627 return 0;

1765HRESULT WASimClient::registerEvent(const RegisteredEvent &eventData)
1766{
1767 if (eventData.code.empty())
1768 return removeEvent(eventData.eventId);
1769
1770 const size_t len = eventData.code.length() + (eventData.name.empty() ? 0 : eventData.name.length() + 1);
1771 if (len >= STRSZ_CMD) {
1772 LOG_ERR << "Resulting command string length " << len << " is greater then maximum size of " << STRSZ_CMD-1 << " bytes.";
1773 return E_INVALIDARG;
1774 }
1775
1776 Private::TrackedEvent *ev = d->findTrackedEvent(eventData.eventId);
1777 if (ev) {
1778 if (!eventData.name.empty() && ev->name != eventData.name) {
1779 LOG_ERR << "Cannot change name of event ID " << eventData.eventId << " after it has been registered.";
1780 return E_INVALIDARG;
1781 }
1782 if (ev->code == eventData.code) {
1783 LOG_INF << "No changes detected in code for event ID " << eventData.eventId << ", skipping update";
1784 return S_OK;
1785 }
1786 ev->code = eventData.code;
1787 }
1788 else {
1789 unique_lock lock(d->mtxEvents);
1790 ev = &d->events.emplace(eventData.eventId, Private::TrackedEvent(eventData.eventId, eventData.code, eventData.name)).first->second;
1791 }
1792 if (isConnected())
1793 ev->sentToServer = SUCCEEDED(d->sendEventRegistration(ev)) || ev->sentToServer;
1794 else
1795 LOG_DBG << "Queued event ID " << eventData.eventId << " for next server connection.";
1796
1797 return S_OK;
1798}

1834{

1836 shared_lock lock(d_const->mtxEvents);

1861HRESULT WASimClient::sendKeyEvent(const std::string &keyEventName, uint32_t v1, uint32_t v2, uint32_t v3, uint32_t v4, uint32_t v5)
1862{
1863 if (keyEventName.empty())
1864 return E_INVALIDARG;
1865
1866 // check the cache first -- this stores both standard key events and custom (mapped) types.
1867 {
1868 shared_lock rdlock(d->mtxKeyEventNames);
1869 const auto pos = d_const->keyEventNameCache.find(keyEventName);
1870 if (pos != d_const->keyEventNameCache.cend())
1871 return sendKeyEvent(pos->second, v1, v2, v3, v4, v5);
1872 // mutex lock and iterator go out of scope
1873 }
1874
1875 // The name wasn't in the cache...
1876 uint32_t keyId;
1877 if (isCustomKeyEventName(keyEventName)) {
1878 // If the event name looks like a "custom" event then try to register it now.
1879 const HRESULT hr = d->registerCustomKeyEvent(keyEventName, &keyId, false);
1880 if FAILED(hr)
1881 return hr;
1882 }
1883 else {
1884 // Otherwise try a lookup() -- this can only work for standard Key Events, and only if we're currently connected.
1885 int32_t id;
1886 const HRESULT hr = lookup(LookupItemType::KeyEventId, keyEventName, &id);
1887 if FAILED(hr)
1888 return hr == E_FAIL ? E_INVALIDARG : hr; // E_FAIL means event name wasn't found at server
1889 keyId = (uint32_t)id;
1890 // save the lookup result
1891 unique_lock rwlock(d->mtxKeyEventNames);
1892 d->keyEventNameCache.insert(std::pair{ keyEventName, keyId });
1893 }
1894
1895 return sendKeyEvent(keyId, v1, v2, v3, v4, v5);
1896}

1929HRESULT WASimClient::list(LookupItemType itemsType)
1930{
1931 // validation
1932 static const set<LookupItemType> validListLookups { LookupItemType::LocalVariable, LookupItemType::DataRequest, LookupItemType::RegisteredEvent };
1933 if (!validListLookups.count(itemsType)) {
1934 LOG_ERR << "Cannot list item of type " << Utilities::getEnumName(itemsType, LookupItemTypeNames) << ".";
1935 return E_INVALIDARG;
1936 }
1937 // TODO: Allow multiple simultaneous list requests for different item types
1938 if (d->listResult.listType != LookupItemType::None) {
1939 LOG_ERR << "A List request for " << Utilities::getEnumName(d->listResult.listType.load(), LookupItemTypeNames) << " is currently pending.";
1940 return E_INVALIDARG;
1941 }
1942 unique_lock lock(d->listResult.mutex);
1943 d->listResult.listType = itemsType;
1944 d->listResult.nextTimeout = Clock::now() + chrono::milliseconds(defaultTimeout());
1945 HRESULT hr;
1946 if FAILED(hr = d->sendServerCommand(Command(CommandId::List, +itemsType), weak_ptr(d->listResult.cv), &d->listResult.token))
1947 return hr;
1948 lock.unlock();
1949 try {
1950 thread(&Private::waitListRequestEnd, d.get()).detach();
1951 }
1952 catch (exception *e) {
1953 LOG_ERR << "Exception trying to schedule thread for List command: " << e->what();
1954 return E_FAIL;
1955 }
1956 return S_OK;
1957}

1959HRESULT WASimClient::lookup(LookupItemType itemType, const std::string &itemName, int32_t *piResult)
1960{
1961 if (itemName.length() >= STRSZ_CMD) {
1962 LOG_ERR << "Item name length " << itemName.length() << " is greater then maximum size of " << STRSZ_CMD-1 << " bytes.";
1963 return E_INVALIDARG;
1964 }
1965 HRESULT hr;
1966 Command response;
1967 if FAILED(hr = d->sendCommandWithResponse(Command(CommandId::Lookup, +itemType, itemName.c_str()), &response))
1968 return hr;
1969 if (response.commandId != CommandId::Ack) {
1970 LOG_WRN << "Lookup command returned Nak response. Reason, if any: " << quoted(response.sData);
1971 return E_FAIL;
1972 }
1973 LOG_DBG << "Lookup: Server returned ID " << (int)response.fData << " and echo name " << quoted(response.sData);
1974 if (piResult)
1975 *piResult = (int)response.fData;
1976 return hr;
1977}
